    const calculateSafeLots = (symbol, riskAmount, stopDistance) => {
      const spec = getContractSpec(symbol);
      const tickValue = spec.tickValue || 1;
      const tickSize = spec.tickSize || 0.01;
      
      // Cálculo: Risco em $ / (Distância do Stop em ticks * Valor por tick)
      let lots = riskAmount / ((stopDistance / tickSize) * tickValue);
      
      // Aplicar limites do usuário e do contrato
      lots = Math.max(spec.minLotSize || 0.01, lots);
      lots = Math.min(lots, aiConfig.maxContracts || 1.0);
      
      // Garantir que não estoure a banca (Margem Segura)
      const maxMarginLots = (currentBalance * 0.5) / (currentPrice * spec.contractSize / 100);
      lots = Math.min(lots, maxMarginLots);
      
      return Number(lots.toFixed(2));
    };
